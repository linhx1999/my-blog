```shell
# elasticsearch:8.12.2

# 运行容器
docker run -d --name es01 -p 9200:9200 -v /usr/share/elasticsearch/data -v /usr/share/elasticsearch/plugins elasticsearch:8.12.2

# 更改密码
docker exec -it es01 /usr/share/elasticsearch/bin/elasticsearch-reset-password -u elastic -i

# 生产kibana token
docker exec -it es01 /usr/share/elasticsearch/bin/elasticsearch-create-enrollment-token -s kibana

# 复制SSL证书
docker cp es01:/usr/share/elasticsearch/config/certs/http_ca.crt .

# 尝试访问
curl --cacert http_ca.crt -u elastic:$ELASTIC_PASSWORD https://localhost:9200
```
