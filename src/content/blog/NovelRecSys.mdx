```shell
# elasticsearch

# 构建docker
docker run -d --name es --net RecSys -p 9200:9200 -e "discovery.type=single-node" \
-e "xpack.security.enrollment.enabled=true" -v es-data:/usr/share/elasticsearch/data \
-v es-plugins:/usr/share/elasticsearch/plugins elasticsearch:8.12.2

# 更改密码
docker exec -it es /usr/share/elasticsearch/bin/elasticsearch-reset-password -u elastic -i

# 设置环境变量
export ELASTIC_PASSWORD="your_password"

# 尝试访问
curl -u elastic:$ELASTIC_PASSWORD http://172.19.6.77:9200
```
