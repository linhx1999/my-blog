```shell
# elasticsearch:8.13.0

# 运行容器
docker run -d --name es -p 9200:9200 -p 9300:9300 -e "discovery.type=single-node" -v es-data:/usr/share/elasticsearch/data -v es-plugins:/usr/share/elasticsearch/plugins elasticsearch:8.13.0

# 更改密码
docker exec -it es /usr/share/elasticsearch/bin/elasticsearch-reset-password -u elastic -i

# 尝试访问
curl -u elastic:$PASSWORD http://172.19.6.77:9200
```
