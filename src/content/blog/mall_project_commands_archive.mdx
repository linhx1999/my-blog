```shell
# mysql 8.4

docker run -p 3306:3306 --name mysql \
-v /volume/mysql/var/lib/mysql:/var/lib/mysql \
-v /volume/mysql/etc/mysql/conf.d:/etc/mysql/conf.d \
-v /volume/mysql/var/log/mysql:/var/log/mysql \
-e MYSQL_ROOT_PASSWORD=root -d mysql:8.4
```

```shell
# redis 7.4
mkdir -p /volume/redis/etc/redis
touch /volume/redis/etc/redis/redis.conf
docker run -p 6379:6379 --name redis \
-v /volume/redis/data:/data \
-v /volume/redis/etc/redis:/etc/redis \
-d redis:7.4 redis-server /etc/redis/redis.conf
```
